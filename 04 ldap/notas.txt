	--env LDAP_ORGANISATION="My Company" \
	--env LDAP_DOMAIN="ldap.example.com" \
	--env LDAP_ADMIN_PASSWORD="StrongAdminPassw0rd" \
	--detach osixia/openldap:latest

	--env LDAP_ORGANISATION="My Company" \
	--env LDAP_DOMAIN="localhost" \
	--env LDAP_ADMIN_PASSWORD="StrongAdminPassw0rd" \
	--detach osixia/openldap:latest -p 80:80
